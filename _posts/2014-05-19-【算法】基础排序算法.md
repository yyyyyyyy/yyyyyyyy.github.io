五种基础排序算法：冒泡、选择、插入、归并、快排Java实现。

```java
package com.yaochow.demo;

import org.junit.Test;

import java.util.Arrays;

public class SortTest {

    /**
     * 冒泡排序
     */
    @Test
    public void bubbleSortTest() {
        int[] array = {1, 9, 10, 8, 40, 36, 24, 7, 4};
        for (int i = 0; i < array.length - 1; i++) {
            for (int j = 0; j < array.length - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    int tmp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = tmp;
                }
            }
        }
        System.out.println(Arrays.toString(array));
    }

    /**
     * 选择排序
     */
    @Test
    public void selectSortTest() {
        int[] array = {1, 9, 10, 8, 40, 36, 24, 7, 4};
        for (int i = 0; i < array.length; i++) {
            int min = i;
            for (int j = i + 1; j < array.length; j++) {
                if (array[j] < array[min]) {
                    min = j;
                }
            }
            if (i != min) {
                int tmp = array[i];
                array[i] = array[min];
                array[min] = tmp;
            }
        }
        System.out.println(Arrays.toString(array));
    }

    /**
     * 插入排序
     */
    @Test
    public void insertSortTest() {
        int[] array = {1, 9, 10, 8, 40, 36, 24, 7, 4};
        for (int i = 1; i < array.length; i++) {
            if (array[i] < array[i - 1]) {
                int tmp = array[i];
                int j;
                for (j = i; j > 0 && array[j - 1] > tmp; j--) {
                    array[j] = array[j - 1];
                }
                array[j] = tmp;
            }
        }
        System.out.println(Arrays.toString(array));
    }

    /**
     * 归并排序
     */
    @Test
    public void mergeSortTest() {
        int[] array = {1, 9, 10, 8, 40, 36, 24, 7, 4};
        int[] tmp = new int[array.length];
        mergeSort(array, tmp, 0, array.length - 1);
        System.out.println(Arrays.toString(array));
    }

    private void mergeSort(int[] array, int[] tmp, int start, int end) {

        if (start >= end) return;

        int mid = (start + end) / 2;
        mergeSort(array, tmp, start, mid);
        mergeSort(array, tmp, mid + 1, end);
        sort(array, tmp, start, mid, end);

    }

    private void sort(int[] array, int[] tmp, int leftPos, int mid, int rightEnd) {

        int rightPos = mid + 1;
        int tmpPos = leftPos;
        int size = rightEnd - leftPos + 1;
        while (leftPos <= mid && rightPos <= rightEnd) {
            if (array[leftPos] <= array[rightPos]) {
                tmp[tmpPos++] = array[leftPos++];
            } else {
                tmp[tmpPos++] = array[rightPos++];
            }
        }

        while (leftPos <= mid) {
            tmp[tmpPos++] = array[leftPos++];
        }

        while (rightPos <= rightEnd) {
            tmp[tmpPos++] = array[rightPos++];
        }

        for (int i = 0; i < size; i++, rightEnd--) {
            array[rightEnd] = tmp[rightEnd];
        }

    }

    /**
     * 快速排序
     */
    @Test
    public void quickSortTest() {
        int[] array = {1, 9, 10, 8, 40, 36, 24, 7, 4};
        quickSort(array, 0, array.length - 1);
        System.out.println(Arrays.toString(array));
    }

    private void quickSort(int[] array, int start, int end) {
        if (start > end) return;

        int left = start;
        int right = end;
        int base = array[start];

        while (left != right) {
            while (left < right && array[right] >= base) {
                right--;
            }
            while (left < right && array[left] <= base) {
                left++;
            }

            if (left != right) {
                int tmp = array[left];
                array[left] = array[right];
                array[right] = tmp;
            }
        }

        array[start] = array[left];
        array[left] = base;
        quickSort(array, start, left - 1);
        quickSort(array, left + 1, end);
    }
}
```
